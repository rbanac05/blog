A bar graph is similar to a histagram, just sideways instead of up and down.

query<- "SELECT name,HR
FROM Teams
WHERE yearID=1980
ORDER BY HR"

result<-sqldf(query)

result$name<-factor(result$name,levels=result$name)

ggplot()+
  geom_bar(data=result,aes(x=name,y=HR),stat='identity',color="blue",fill="green")+
  coord_flip()+
  ylab("Home Runs")+
  xlab("Teams")+
  ggtitle("1980 Home Run Distribution")

A histogram is a visual representation of data using one column.

query<-"SELECT weight
FROM Master"

result<-sqldf(query)

ggplot()+
  geom_histogram(data=result,aes(x=weight),color="blue",fill="green",bins=50)+
  ggtitle("Body Weight Distribution For Baseball Players")

Scatter Plots

library(sqldf)
library(Lahman)
library(ggplot2)

A scatter plot is a data representation through dots.

#Extractcing Data----------------------------

query<-"SELECT playerID,sum(HR) AS career_HR ,sum(SO) AS career_SO 
FROM Batting
GROUP BY playerID
HAVING sum(HR)>=400"

result<-sqldf(query)

#Visulaizing Data----------------------------

ggplot()+
  geom_point(data=result,aes(x=career_SO,y=career_HR), size=3, color="blue")+
  ggtitle("Career Strikeouts vs. Homeruns for Great Hitters")+
  xlab("Career Strikeouts")+
  ylab("Career Homerunes")

Time Series

library(sqldf)
library(Lahman)
library(ggplot2)

Time series are similar to scatter plots in that you can have dots on the graph, but we mostly look at the line showing the trend of data.

query<- "SELECT yearID, HR
FROM Batting
WHERE playerID='ruthba01'"
result<-sqldf(query)

ggplot()+
  geom_line(data=result,aes(x=yearID,y=HR))+
  ggtitle("Ruth's Home Runs of the Years")+
  xlab("Year")+
  ylab("Home Runs")
